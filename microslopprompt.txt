### `routing/profiles.py`

```python
class NetworkProfile:
    def __init__(self, id, dns_resolver, interface, tailscale_exit, proxy=None):
        self.id = id
        self.dns_resolver = dns_resolver
        self.interface = interface
        self.tailscale_exit = tailscale_exit
        self.proxy = proxy

def build_profiles(concurrency: int, resolvers: list, tailscale_state) -> list[NetworkProfile]:
    # deterministic or random assignment of resolver + route per slot
    profiles = []
    for i in range(concurrency):
        resolver = select_resolver(i, resolvers)
        exit_node = select_exit_node(i, tailscale_state)
        iface = select_interface(exit_node)
        profiles.append(NetworkProfile(i, resolver, iface, exit_node))
    return profiles
```

### `crawler/scheduler.py`
```python
from concurrent.futures import ThreadPoolExecutor

def schedule_crawl(targets, profiles, settings):
    with ThreadPoolExecutor(max_workers=len(profiles)) as pool:
        futures = []
        for profile, target in zip(profiles, targets):
            futures.append(pool.submit(run_worker, profile, target, settings))
        for f in futures:
            f.result()
```

### `crawler/worker.py`
```python
from .fetch import fetch_page
from .parse import extract_links
from .snapshot_pdf import save_pdf_snapshot

def run_worker(profile, target, settings):
    # 1. apply DNS for this worker
    apply_resolver(profile.dns_resolver)
    try:
        # 2. create HTTP client bound to profile.interface / proxy
        session = build_http_client(profile)

        # 3. obey robots.txt
        if not allowed_by_robots(session, target):
            return

        # 4. fetch
        html, response_meta = fetch_page(session, target)

        # 5. snapshot to PDF
        pdf_path = save_pdf_snapshot(target, html, settings.PDF_OUTPUT_ROOT, profile)

        # 6. parse links (optional further queueing)
        links = extract_links(html, base_url=target)

        # 7. persist crawl record
        store_crawl_record(
            url=target,
            html=html,
            pdf_path=pdf_path,
            profile_id=profile.id,
            dns_resolver=profile.dns_resolver,
            route_meta=response_meta,
        )
    finally:
        restore_original_dns()
```

### `core/orchestrator.py`
```python
from system_profile.collector import collect_system_profile
from dns.transplant import load_resolvers
from routing.tailscale_manager import get_tailscale_state
from routing.profiles import build_profiles
from remote.client import RemoteQueueClient
from crawler.scheduler import schedule_crawl
from storage.db import load_crawl_records
from analytics.session_context import build_session_context

def run_session(settings):
    # 1. system profile → CSV
    system_csv = collect_system_profile(settings)

    # 2. optional remote coordination
    remote_client = RemoteQueueClient(...) if settings.REMOTE_ENABLED else None
    if remote_client:
        remote_client.register_session(system_csv)

    # 3. load DNS resolvers + tailscale state
    resolvers = load_resolvers(settings.DNS_PROFILE_FILE)
    tailscale_state = get_tailscale_state()

    # 4. build network profiles per thread
    profiles = build_profiles(settings.MAX_CONCURRENCY, resolvers, tailscale_state)

    # 5. get targets (local list or remote queue)
    if remote_client:
        targets = remote_client.fetch_assignments(len(profiles))
    else:
        targets = load_local_targets()

    # 6. schedule crawl with per-profile split routing
    schedule_crawl(targets, profiles, settings)

    # 7. gather crawl records + build final context
    records = load_crawl_records()
    context = build_session_context(system_csv, records)
    return context
```
